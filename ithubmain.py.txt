import java.util.Scanner;


public class Main {
    private static Scanner scanner = new Scanner(System.in);
    private static String appName = "Улучшенное приложение";
    

    public static void main(String[] args) {
        System.out.println("=== " + appName + " ===");
        System.out.println("Версия: 1.1.1");
        System.out.println("Статус: Новые функции + исправления ошибок");
        
        startInteractiveMode();
    }
    
   
    public static void startInteractiveMode() {
        System.out.println("\n--- Интерактивный режим ---");
        
        while (true) {
            displayMenu();
            String choice = scanner.nextLine();
            
            switch (choice) {
                case "1":
                    performAdvancedOperation();
                    break;
                case "2":
                    showSettings();
                    break;
                case "3":
                    displayVersionInfo();
                    break;
                case "0":
                    System.out.println("Выход из приложения...");
                    return;
                default:
                    System.out.println("Неверный выбор! Попробуйте снова.");
            }
        }
    }
    
 
    public static void displayMenu() {
        System.out.println("\nВыберите действие:");
        System.out.println("1. Выполнить расширенную операцию");
        System.out.println("2. Настройки");
        System.out.println("3. Информация о версии");
        System.out.println("0. Выход");
        System.out.print("Ваш выбор: ");
    }
    

    public static void performAdvancedOperation() {
        System.out.println("\n--- Расширенная операция ---");

        int a = scanner.nextInt();

        scanner.nextLine();
        
        int sum = a + b;
        int product = a * b;
        
        System.out.println("Сумма: " + sum);
        System.out.println("Произведение: " + product);
        System.out.println("Операция завершена успешно!");

import java.util.Scanner;
import java.util.HashMap;
import java.util.Map;


public class Main {
    private static Application app;
    
    public static void main(String[] args) {
        app = new Application("Многофункциональное приложение", "2.0.0");
        app.start();
    }
}


class Application {
    private String name;
    private String version;
    private Scanner scanner;
    private Map<String, String> languagePack;
    private String currentLanguage;
    
    public Application(String name, String version) {
        this.name = name;
        this.version = version;
        this.scanner = new Scanner(System.in);
        this.languagePack = new HashMap<>();
        this.currentLanguage = "ru";
        initializeLanguagePack();
    }
    
 
    private void initializeLanguagePack() {

        languagePack.put("ru.welcome", "Добро пожаловать в ");
        languagePack.put("ru.version", "Версия: ");
        languagePack.put("ru.menu.title", "Главное меню");
        languagePack.put("ru.menu.option1", "Расширенные вычисления");
        languagePack.put("ru.menu.option2", "Языковые настройки");
        languagePack.put("ru.menu.option3", "Информация о системе");
        languagePack.put("ru.menu.option4", "Экспорт данных");
        languagePack.put("ru.menu.exit", "Выход");
        languagePack.put("ru.goodbye", "До свидания!");
        

        languagePack.put("en.welcome", "Welcome to ");
        languagePack.put("en.version", "Version: ");
        languagePack.put("en.menu.title", "Main Menu");
        languagePack.put("en.menu.option1", "Advanced Calculations");
        languagePack.put("en.menu.option2", "Language Settings");
        languagePack.put("en.menu.option3", "System Information");
        languagePack.put("en.menu.option4", "Data Export");
        languagePack.put("en.menu.exit", "Exit");
        languagePack.put("en.goodbye", "Goodbye!");
    }
    

    public void start() {
        displayHeader();
        mainLoop();
        shutdown();
    }
    

    private void displayHeader() {
        System.out.println("=== " + name + " ===");
        System.out.println(getLocalizedString("version") + version);
        System.out.println(getLocalizedString("welcome") + name + "!");
    }
    

    private void mainLoop() {
        while (true) {
            displayMainMenu();
            String choice = scanner.nextLine();
            
            switch (choice) {
                case "1":
                    new CalculationModule().execute();
                    break;
                case "2":
                    changeLanguage();
                    break;
                case "3":
                    new SystemInfoModule().displayInfo(version);
                    break;
                case "4":
                    new ExportModule().exportData();
                    break;
                case "0":
                    return;
                default:
                    System.out.println("Неверный выбор! Попробуйте снова.");
            }
        }
    }
    

    private void displayMainMenu() {
        System.out.println("\n--- " + getLocalizedString("menu.title") + " ---");
        System.out.println("1. " + getLocalizedString("menu.option1"));
        System.out.println("2. " + getLocalizedString("menu.option2"));
        System.out.println("3. " + getLocalizedString("menu.option3"));
        System.out.println("4. " + getLocalizedString("menu.option4"));
        System.out.println("0. " + getLocalizedString("menu.exit"));
        System.out.print("Ваш выбор: ");
    }
    
    /**
     * Получает локализованную строку
     * Новая функция в версии 2.0.0
     */
    private String getLocalizedString(String key) {
        return languagePack.getOrDefault(currentLanguage + "." + key, 
                                       languagePack.get("ru." + key));
    }
    

    private void changeLanguage() {
        System.out.println("\n--- Языковые настройки ---");
        System.out.println("1. Русский");
        System.out.println("2. English");
        System.out.print("Выберите язык: ");
        
        String choice = scanner.nextLine();
        if (choice.equals("2")) {
            currentLanguage = "en";
            System.out.println("Language changed to English!");
        } else {
            currentLanguage = "ru";
            System.out.println("Язык изменен на русский!");
        }
    }
    

    private void shutdown() {
        System.out.println("\n" + getLocalizedString("goodbye"));
        scanner.close();
    }
}


class CalculationModule {
    public void execute() {
        System.out.println("\n--- Расширенные вычисления ---");
        Scanner scanner = new Scanner(System.in);
        
        System.out.print("Введите первое число: ");
        double a = scanner.nextDouble();
        System.out.print("Введите второе число: ");
        double b = scanner.nextDouble();
        
        System.out.println("Результаты:");
        System.out.println("Сумма: " + (a + b));
        System.out.println("Разность: " + (a - b));
        System.out.println("Произведение: " + (a * b));
        System.out.println("Частное: " + (b != 0 ? a / b : "Ошибка: деление на ноль"));
    }
}


class SystemInfoModule {
    public void displayInfo(String version) {
        System.out.println("\n--- Информация о системе ---");
        System.out.println("Версия приложения: " + version);
        System.out.println("Версия Java: " + System.getProperty("java.version"));
        System.out.println("Архитектура: " + System.getProperty("os.arch"));
        System.out.println("ОС: " + System.getProperty("os.name"));
    }
}
